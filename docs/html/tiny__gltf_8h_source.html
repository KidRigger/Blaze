335  return (it != object_value_.end()) ? it->second : null_value;

756  int componentSizeInBytes =

758  if (componentSizeInBytes <= 0) {

763  if (typeSizeInBytes <= 0) {
764  return -1;
765  }
766

1379  static bool Equals(const tinygltf::Value& one, const tinygltf::Value& other) {
1380  if (one.Type() != other.Type()) return false;
1381 
1382  switch (one.Type()) {
1383  case NULL_TYPE:
1384  return true;
1385  case BOOL_TYPE:
1386  return one.Get<bool>() == other.Get<bool>();
1387  case REAL_TYPE:
1388  return TINYGLTF_DOUBLE_EQUAL(one.Get<double>(), other.Get<double>());
1389  case INT_TYPE:
1390  return one.Get<int>() == other.Get<int>();
1391  case OBJECT_TYPE: {
1392  auto oneObj = one.Get<tinygltf::Value::Object>();
1393  auto otherObj = other.Get<tinygltf::Value::Object>();
1394  if (oneObj.size() != otherObj.size()) return false;

1396  auto otherIt = otherObj.find(it.first);
1397  if (otherIt == otherObj.end()) return false;
1398 
1399  if (!Equals(it.second, otherIt->second)) return false;
1400  }
1401  return true;
1402  }
1403  case ARRAY_TYPE: {
1404  if (one.Size() != other.Size()) return false;
1405  for (int i = 0; i < int(one.Size()); ++i)
1406  if (!Equals(one.Get(i), other.Get(i))) return false;
1407  return true;
1408  }
1409  case STRING_TYPE:
1410  return one.Get<std::string>() == other.Get<std::string>();
1411  case BINARY_TYPE:

1413  other.Get<std::vector<unsigned char> >();
